import javax.persistence.Id;

@Entity
public class glass {
    @Id
    @GeneratedValue(strategy= GenerationType.TABLE)
    private int id;
    ///attributes///

    private String type;
    private int volume;

    ///constructors///
    public glass(String type,int volume) {
        this.type = type;
        this.volume = volume;
